<script>

    let queue = JSON.parse(localStorage.getItem('gameQueue')) || [];

    let history = JSON.parse(localStorage.getItem('gameHistory')) || [];

    let stats = JSON.parse(localStorage.getItem('gameStats')) || { done: 0, backlogged: 0 };

    function updateUI() {

        document.getElementById('stat-queue').innerText = `Queue: ${queue.length}`;

        document.getElementById('stat-done').innerText = `Done: ${stats.done}`;

        document.getElementById('stat-backlog').innerText = `Backlog: ${stats.backlogged}`;

        const titleEl = document.getElementById('game-title');

        const platformTag = document.getElementById('tag-platform');

        const playtimeTag = document.getElementById('tag-playtime');

        if (queue.length > 0) {

            const game = queue[0];

            titleEl.innerText = game.name;

            titleEl.classList.remove('empty-state');

            if (game.platform) {

                platformTag.innerText = game.platform;

                platformTag.className = 'tag visible ' + (['Steam', 'Epic', 'PlayStation', 'Xbox', 'GOG'].includes(game.platform) ? game.platform : 'other');

            } else {

                platformTag.className = 'tag';

            }

            if (game.playtime !== undefined && game.playtime !== null) {

                playtimeTag.innerText = `${game.playtime} hrs`;

                playtimeTag.classList.add('visible');

            } else {

                playtimeTag.classList.remove('visible');

            }

        } else {

            titleEl.innerText = "You're all caught up! Import more games.";

            titleEl.classList.add('empty-state');

            platformTag.className = 'tag';

            playtimeTag.classList.remove('visible');

        }

        saveData();

    }

    function handleAction(action) {

        if (queue.length === 0) return;

       

        let currentGame = queue.shift();

        history.push({ game: currentGame, action: action });

        if (action === 'skip') {

            queue.push(currentGame);

        } else if (action === 'backlog') {

            stats.backlogged++;

        } else if (action === 'done') {

            stats.done++;

        }

        updateUI();

    }

    function undoAction() {

        if (history.length === 0) return alert("Nothing to undo!");

       

        let lastEvent = history.pop();

       

        if (lastEvent.action === 'skip') {

            queue.pop();

        } else if (lastEvent.action === 'backlog') {

            stats.backlogged--;

        } else if (lastEvent.action === 'done') {

            stats.done--;

        }

        queue.unshift(lastEvent.game);

        updateUI();

    }

    function importCSV(event) {

        const file = event.target.files[0];

        if (!file) return;

        const reader = new FileReader();

        reader.onload = function(e) {

            const text = e.target.result;

            const lines = text.split('\n');

            let addedCount = 0;

            for (let i = 1; i < lines.length; i++) {

                if (!lines[i].trim()) continue;

                // Matches format: "Name","Source","Platforms","Status","PlaytimeHours"

                const match = lines[i].match(/^"([^"]+)","([^"]*)","([^"]*)","([^"]*)","([^"]*)"/);

                if (match) {

                    const gameName = match[1];

                    const gameSource = match[2];

                    const gamePlaytime = match[5];

                   

                    if (!queue.some(g => g.name === gameName)) {

                        queue.push({

                            name: gameName,

                            platform: gameSource || 'Unknown',

                            playtime: parseFloat(gamePlaytime) || 0

                        });

                        addedCount++;

                    }

                }

            }

            alert(`Successfully imported ${addedCount} new games!`);

            updateUI();

        };

        reader.readAsText(file);

    }

    function exportJSON() {

        const data = { queue, history, stats };

        const blob = new Blob([JSON.stringify(data, null, 2)], {type: "application/json"});

        triggerDownload(blob, 'backlog_tracker_backup.json');

    }

    function exportCSV() {

        if (history.length === 0) return alert("No history to export.");

        let csvContent = "Name,Decision,Platform,PlaytimeHours\n";

        history.forEach(item => {

            if (item.action !== 'skip') {

                csvContent += `"${item.game.name}","${item.action}","${item.game.platform}","${item.game.playtime}"\n`;

            }

        });

        const blob = new Blob([csvContent], {type: "text/csv"});

        triggerDownload(blob, 'backlog_results.csv');

    }

    function triggerDownload(blob, filename) {

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url;

        a.download = filename;

        a.click();

        URL.revokeObjectURL(url);

    }

    function clearData() {

        if(confirm("Are you sure you want to delete all progress?")) {

            queue = [];

            history = [];

            stats = { done: 0, backlogged: 0 };

            updateUI();

        }

    }

    function saveData() {

        localStorage.setItem('gameQueue', JSON.stringify(queue));

        localStorage.setItem('gameHistory', JSON.stringify(history));

        localStorage.setItem('gameStats', JSON.stringify(stats));

    }

    updateUI();

</script>
